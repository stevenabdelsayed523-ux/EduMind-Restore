// ════════════════════════════════════════
//  UNIVERSAL AI CALL

async function callAI(prompt, maxT=4000){
  const cfg = lsg('ai_config', null);
  if(!cfg || !cfg.key) throw new Error('NO_KEY');
  const {provider, key, model, customUrl} = cfg;
  let url, headers, bodyObj, extractFn;

  if(provider === 'gemini'){
    url = 'https://generativelanguage.googleapis.com/v1beta/models/'+model+':generateContent?key='+key;
    headers = {'Content-Type':'application/json'};
    bodyObj = {contents:[{parts:[{text:prompt}]}], generationConfig:{maxOutputTokens:maxT, temperature:0.7}};
    extractFn = function(d){ return d.candidates[0].content.parts.map(function(p){return p.text||'';}).join(''); };
  } else if(provider === 'anthropic'){
    url = 'https://api.anthropic.com/v1/messages';
    headers = {'Content-Type':'application/json','x-api-key':key,'anthropic-version':'2023-06-01','anthropic-dangerous-direct-browser-access':'true'};
    bodyObj = {model:model, max_tokens:maxT, messages:[{role:'user',content:prompt}]};
    extractFn = function(d){ return d.content.map(function(b){return b.text||'';}).join(''); };
  } else if(provider === 'cohere'){
    url = 'https://api.cohere.com/v1/chat';
    headers = {'Content-Type':'application/json','Authorization':'Bearer '+key};
    bodyObj = {model:model, message:prompt, max_tokens:maxT};
    extractFn = function(d){ return d.text || d.message || ''; };
  } else {
    // OpenAI-compatible: openai / groq / mistral / openrouter / together / custom
    const baseUrls = {
      openai:'https://api.openai.com/v1',
      groq:'https://api.groq.com/openai/v1',
      mistral:'https://api.mistral.ai/v1',
      openrouter:'https://openrouter.ai/api/v1',
      together:'https://api.together.xyz/v1',
      custom: customUrl || 'http://localhost:11434/v1'
    };
    url = (baseUrls[provider] || baseUrls.custom) + '/chat/completions';
    headers = {'Content-Type':'application/json','Authorization':'Bearer '+key};
    if(provider==='openrouter') headers['HTTP-Referer'] = location.href;
    bodyObj = {model:model, max_tokens:maxT, temperature:0.7, messages:[{role:'user',content:prompt}]};
    extractFn = function(d){ return d.choices[0].message.content || ''; };
  }

  let r;
  try {
    r = await fetch(url, {method:'POST', headers:headers, body:JSON.stringify(bodyObj)});
  } catch(netErr) {
    throw new Error('Network error: ' + netErr.message);
  }
  const body = await r.json().catch(function(){return {};});
  if(!r.ok){
    const msg = (body && body.error && (body.error.message || body.error)) || JSON.stringify(body);
    if(r.status===401 || r.status===403) throw new Error('INVALID_KEY');
    if(r.status===429) throw new Error('RATE_LIMIT');
    throw new Error(msg || 'HTTP '+r.status);
  }
  try {
    const text = extractFn(body);
    if(!text) throw new Error('empty response');
    return text;
  } catch(e) {
    console.error('AI response parsing failed:', JSON.stringify(body));
    throw new Error('Could not parse AI response. Check console (F12).');
  }
}

function parseJSON(text){
  text = text.trim();
  text = text.replace(/^```(?:json)?\s*/i,'').replace(/\s*```\s*$/,'').trim();
  const arrMatch = text.match(/(\[[\s\S]*\])/);
  if(arrMatch) return JSON.parse(arrMatch[1]);
  return JSON.parse(text);
}

function renderQDots(){
  document.getElementById('qdots').innerHTML=S.questions.map((q,i)=>{
    let c='qdot';
    if(i===S.cur)c+=' cur';
    else if(S.answers[i]!==undefined)c+=S.answers[i]===q.correct?' correct':' wrong';
    else c+=' answered'; // just answered
    if(S.answers[i]===undefined&&i!==S.cur)c=c.replace(' answered','');
    return `<div class="${c}" onclick="jumpQ(${i})" title="Q${i+1}"></div>`;
  }).join('');
}

function finishTest(){
  if(S.testTimerIntv){clearInterval(S.testTimerIntv);S.testTimerIntv=null;}
  const elapsed=Math.round((Date.now()-S.testStart)/1000);
  const total=S.questions.length;
  let correct=0;S.questions.forEach((q,i)=>{if(S.answers[i]===q.correct)correct++;});
  const pct=Math.round(correct/total*100);
  // Stats
  if(pct===100)lss('perfect',(lsg('perfect',0))+1);
  if(pct>(lsg('best_'+(S.subject||''),0)||0))lss('best_'+(S.subject||''),pct);
  const rec=lsg('recent',[]);
  rec.unshift({subject:S.subject,year:S.year,diff:S.diff,score:pct,date:new Date().toLocaleDateString(),correct,total});
  lss('recent',rec.slice(0,30));
  lss('focus',(lsg('focus',0))+Math.ceil(elapsed/60));
  lss('tests',(lsg('tests',0))+1);
  logCalSession();
  // XP
  const xpEarned=Math.round(correct*5+(pct/100)*50);
  lss('xp',(lsg('xp',0))+xpEarned);
  // Score ring
  const off=364-(364*pct/100);
  const rc=document.getElementById('rCirc');if(rc)rc.style.strokeDashoffset=off;
  const sp=document.getElementById('sPct');if(sp)sp.textContent=pct+'%';
  const vd=pct>=90?'Outstanding! 🎉':pct>=70?'Great work! 👏':pct>=50?'Good effort! 💪':'Keep going! 📚';
  const sv=document.getElementById('sVerdict');if(sv)sv.textContent=vd;
  const ss=document.getElementById('sSub');if(ss)ss.textContent=pct>=70?'Solid result — review the ones you missed.':'Review the explanations then try again!';
  const ro=document.getElementById('rOk');if(ro)ro.textContent=correct;
  const rb=document.getElementById('rBad');if(rb)rb.textContent=total-correct;
  const m=Math.floor(elapsed/60),sc=elapsed%60;
  const rt=document.getElementById('rTime');if(rt)rt.textContent=String(m).padStart(2,'0')+':'+String(sc).padStart(2,'0');
  const rx=document.getElementById('rXP');if(rx)rx.textContent='+'+xpEarned+' XP';
  const L=['A','B','C','D'];
  const rl=document.getElementById('resList');
  if(rl)rl.innerHTML=S.questions.map((q,i)=>{
    const ua=S.answers[i],ok=ua===q.correct;
    return '<div class="ri '+(ok?'ok':'bad')+'"><div class="ri-ico">'+(ok?'✅':'❌')+'</div><div><div class="ri-q">'+(i+1)+'. '+q.q+'</div><div class="ri-a" style="color:'+(ok?'#6ee7b7':'#fca5a5')+'">'+(ua!==undefined?L[ua]+'. '+q.choices[ua]:'Not answered')+'</div>'+((!ok&&ua!==undefined)?'<div class="ri-a" style="color:#6ee7b7;margin-top:3px">✓ '+L[q.correct]+'.   }).join('');
  if(pct>=80)launchConfetti(pct>=90?120:60);
  showScreen('sResults');aiFabShow(false);

}

// Detect provider from API key prefix/format
function detectProviderFromKey(key){
  if(!key || key.length < 6) return null;
  const k = key.trim();
  if(k.startsWith('AIza'))                         return 'gemini';
  if(k.startsWith('sk-ant-'))                       return 'anthropic';
  if(k.startsWith('sk-or-'))                        return 'openrouter';
  if(k.startsWith('gsk_'))                          return 'groq';
  if(k.startsWith('sk-') && k.length > 40)          return 'openai';
  if(/^[A-Za-z0-9]{32,}$/.test(k) && k.length===32) return 'cohere';   // Cohere keys are 32-char alphanumeric
  if(/^[A-Za-z0-9_-]{32,}$/.test(k) && k.length>40 && !k.startsWith('sk')) return 'mistral';
  return null;
}

function autoDetectKey(val){
  const detected = detectProviderFromKey(val);
  const badge = document.getElementById('autoDetectBadge');
  const status = document.getElementById('apiKeyStatus');
  status.style.display='none';
  if(detected){
    badge.style.display='block';
    badge.textContent='⚡ ' + (PROVIDERS[detected]?.name || detected) + ' detected';
    // Switch to detected provider silently (keep current key, update UI)
    document.querySelectorAll('.provider-card').forEach(c=>c.classList.toggle('on', c.dataset.p===detected));
    const cfg = PROVIDERS[detected];
    // update info
    const getLink = cfg.keyLink ? `<a href="${cfg.keyLink}" target="_blank" style="color:var(--cobalt)">${cfg.keyLinkLabel} ↗</a>` : '';
    document.getElementById('providerInfo').innerHTML = cfg.info + (getLink ? ' Get your key: ' + getLink : '');
    // update models
    const ms = document.getElementById('modelSelect');
    ms.innerHTML = cfg.models.map(m=>`<option value="${m.id}">${m.label}</option>`).join('');
    // restore saved model if same provider
    const saved = lsg('ai_config',null);
    if(saved && saved.provider===detected && saved.model) ms.value = saved.model;
    document.getElementById('customUrlRow').style.display = 'none';
  } else {
    badge.style.display='none';
  }
}

function selectProvider(p){
  document.querySelectorAll('.provider-card').forEach(c=>c.classList.toggle('on', c.dataset.p===p));
  const cfg = PROVIDERS[p];
  const info = document.getElementById('providerInfo');
  const getLink = cfg.keyLink ? `<a href="${cfg.keyLink}" target="_blank" style="color:var(--cobalt)">${cfg.keyLinkLabel} ↗</a>` : '';
  info.innerHTML = cfg.info + (getLink ? ' Get your key: ' + getLink : '');
  // models
  const ms = document.getElementById('modelSelect');
  ms.innerHTML = cfg.models.map(m=>`<option value="${m.id}">${m.label}</option>`).join('');
  // custom url
  document.getElementById('customUrlRow').style.display = p==='custom'?'block':'none';
  document.getElementById('keyLabel').textContent = p==='custom'?'API Key (leave blank for no auth)':'API Key';
  // restore saved values for this provider, or clear

  }

  autoDetectKey(inp.value);
  document.getElementById('apiKeyStatus').style.display='none';
}

async function saveAPIKey(){
  const key = document.getElementById('apiKeyInput').value.trim();
  // Use the currently highlighted provider card (could have been auto-switched)
  const activeP = document.querySelector('.provider-card.on')?.dataset.p || 'gemini';
  const model = document.getElementById('modelSelect').value;
  const customUrl = document.getElementById('customUrl').value.trim();
  const status = document.getElementById('apiKeyStatus');

  const show = (msg, ok, neutral) => {
    status.style.display='block';
    if(neutral){ status.style.background='var(--glass)'; status.style.border='1px solid var(--rim)'; status.style.color='var(--fog)'; }
    else {
      status.style.background = ok ? 'rgba(16,201,138,.1)' : 'rgba(244,63,94,.1)';
      status.style.border = ok ? '1px solid rgba(16,201,138,.25)' : '1px solid rgba(244,63,94,.25)';
      status.style.color = ok ? '#6ee7b7' : '#fca5a5';
    }
    status.textContent = msg;
  };

  if(!key && activeP !== 'custom'){ show('Please paste your API key.', false); return; }

  // Double-check auto-detect agrees or give a hint
  const detected = detectProviderFromKey(key);
  if(detected && detected !== activeP){
    show(`⚡ This looks like a ${PROVIDERS[detected]?.name||detected} key — switching provider.`, true);
    selectProvider(detected);
    await new Promise(r=>setTimeout(r,600));
  }

  show('Testing connection…', true, true);
  const finalP = document.querySelector('.provider-card.on')?.dataset.p || activeP;
  const finalModel = document.getElementById('modelSelect').value;
  const newCfg = {provider:finalP, key, model:finalModel, customUrl};
  lss('ai_config', newCfg);

  try{
    await callAI('Say "ok" in one word.', 10);
    show('✓ Connected to ' + (PROVIDERS[finalP]?.name||finalP) + '! Ready to generate.', true);
    updateKeyBadge();
    setTimeout(closeAPIKey, 1800);
  }catch(e){
    lss('ai_config', null);
    if(e.message==='INVALID_KEY') show('✗ Key rejected — check it is correct for ' + (PROVIDERS[finalP]?.name||finalP) + '.', false);
    else if(e.message==='RATE_LIMIT'){ lss('ai_config',newCfg); show('⚠ Rate limited but key saved — try generating in a moment.', true); updateKeyBadge(); setTimeout(closeAPIKey,2000); }
    else show('✗ ' + e.message, false);
  }
}

async function genNotes(){
  const topic=document.getElementById('notesTopicIn')?.value.trim();
  if(!topic){alert('Enter a topic first.');return;}
  if(!lsg('ai_cfg',null)){openAPIKey();return;}
  const incl=[];
  if(document.getElementById('chkDef')?.checked)incl.push('key definitions');
  if(document.getElementById('chkEx')?.checked)incl.push('examples');
  if(document.getElementById('chkFml')?.checked)incl.push('formulas');
  if(document.getElementById('chkTip')?.checked)incl.push('exam tips');
  const btn=document.getElementById('genNotesBtn');if(btn){btn.disabled=true;btn.textContent='Generating…';}
  const prompt='Create '+_notesDetail+' study notes on: "'+topic+'" for '+(S.year||'secondary school')+' students. Include: '+incl.join(', ')+'. Return ONLY valid JSON: {"title":"...","sections":[{"heading":"...","content":"...","bullets":["..."]}],"examTips":["..."],"keyTerms":[{"term":"...","definition":"..."}]}. No markdown.';
  try{
    const text=await callAI(prompt,3000);
    const data=JSON.parse(text.replace(/^```(?:json)?\s*/i,'').replace(/\s*```\s*$/,'').trim());
    const nt=document.getElementById('notesTopic');if(nt)nt.textContent=data.title||topic;
    let html2='';
    (data.sections||[]).forEach(s=>{html2+='<div style="margin-bottom:1.4rem"><h4 style="font-size:.88rem;font-weight:700;color:var(--cobalt);text-transform:uppercase;letter-spacing:.08em;margin-bottom:.55rem">'+s.heading+'</h4>';if(s.content)html2+='<p style="font-size:.88rem;line-height:1.7;color:var(--snow);margin-bottom:.5rem">'+s.content+'</p>';if(s.bullets?.length)html2+='<ul style="padding-left:1.15rem">'+s.bullets.map(b=>'<li style="font-size:.88rem;line-height:1.7;color:var(--snow);margin-bottom:.28rem">'+b+'</li>').join('')+'</ul></div>';});
    if(data.keyTerms?.length)html2+='<div style="margin-bottom:1.4rem"><h4 style="font-size:.88rem;font-weight:700;color:var(--cobalt);text-transform:uppercase;letter-spacing:.08em;margin-bottom:.55rem">Key Terms</h4><ul style="padding-left:1.15rem">'+data.keyTerms.map(t=>'<li style="font-size:.88rem;line-height:1.7;color:var(--snow);margin-bottom:.28rem"><b>'+t.term+'</b> — '+t.definition+'</li>').join('')+'</ul></div>';
    if(data.examTips?.length)html2+='<div style="margin-bottom:1.4rem"><h4 style="font-size:.88rem;font-weight:700;color:var(--cobalt);text-transform:uppercase;letter-spacing:.08em;margin-bottom:.55rem">Exam Tips</h4><ul style="padding-left:1.15rem">'+data.examTips.map(t=>'<li style="font-size:.88rem;line-height:1.7;color:var(--snow);margin-bottom:.28rem">'+t+'</li>').join('')+'</ul></div>';
    const nb=document.getElementById('notesBody');if(nb)nb.innerHTML=html2;
    const nr=document.getElementById('notesResult');if(nr)nr.style.display='block';
    lss('notes_done',(lsg('notes_done',0))+1);checkAch();
  }catch(e){if(e.message==='NO_KEY'||e.message==='INVALID_KEY')openAPIKey();else alert('Error: '+e.message);}
  if(btn){btn.disabled=false;btn.textContent='Generate Notes →';}
}
